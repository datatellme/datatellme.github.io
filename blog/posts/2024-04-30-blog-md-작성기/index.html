    blog.md 문서작성기

    아래는 수정된 코드입니다. 주요 변경 사항은 current_time을 output_file 이전에 정의하고, output_file에 파일 확장자를 추가한 것입니다. 또한 output_directory를 생성하는 부분도 약간 수정되었습니다.

    import tkinter as tk
from tkinter import messagebox
import os
from datetime import datetime

def save_text_to_file():
    text_content = text_entry.get("1.0", tk.END)  # 텍스트 입력란의 내용 가져오기

    # 현재 시간
    current_time = datetime.now().strftime("%Y-%m-%d")

    # 출력 디렉토리 및 파일 경로 설정
    output_directory = r'C:\gppj\data-tell-me_ver02\docs\blog\posts'
    if not os.path.exists(output_directory):
        os.makedirs(output_directory)
    output_file = f'{current_time}-first-post.md'
    output_path = os.path.join(output_directory, output_file)

    # Rmd 문서 내용 생성
    rmd_content = f'''---
title: "My First Blog Post"
date: "{current_time}"
categories:
  - 팬시
---
{text_content}
'''

    # Rmd 파일에 저장
    with open(output_path, 'w', encoding='utf-8') as file:
        file.write(rmd_content)

    # 저장 완료 메시지 출력
    messagebox.showinfo("저장 완료", "md 문서가 저장되었습니다.")

# Tkinter GUI 생성
root = tk.Tk()
root.title("md 문서 작성기")
root.geometry("400x500")

# 텍스트 입력 위젯 생성
text_entry = tk.Text(root)
text_entry.pack(fill="both", expand=True)

# 저장 버튼 생성
save_button = tk.Button(root, text="저장", command=save_text_to_file)
save_button.pack()

root.mainloop()

    이제 코드가 원하는 대로 작동할 것입니다! 😊👍 만약 추가 질문이 있거나 더 도움이 필요하시면 언제든지 말씀해주세요!
