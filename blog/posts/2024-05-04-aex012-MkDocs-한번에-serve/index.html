    • A01: 아래는 주어진 정보를 사용하여 MkDocs을 통해 Markdown 파일을 HTML로 변환하고 로컬 서버를 시작하는 코드입니다.
    import os
from mkdocs.commands.build import build
from mkdocs.config import load_config
import subprocess

def convert_md_to_html(md_directory, html_directory, mkdocs_config_path):
    try:
        # MkDocs 설정 로드
        config = load_config(mkdocs_config_path)

        # HTML 파일이 생성될 디렉토리 생성
        if not os.path.exists(html_directory):
            os.makedirs(html_directory)

        # 현재 작업 디렉토리 설정
        os.chdir(md_directory)

        # HTML 파일로 변환
        build(config)

        # 로컬 서버 시작
        subprocess.run(['mkdocs', 'serve', '-f', mkdocs_config_path])
    except FileNotFoundError:
        print("MkDocs를 찾을 수 없습니다. MkDocs가 설치되어 있는지 확인하세요.")
    except Exception as e:
        print(f"오류 발생: {e}")

if __name__ == "__main__":
    # 디렉토리 및 파일 경로 설정
    md_directory = r"C:\gppj\data-tell-me_ver02\docs"
    html_directory = r"C:\gppj\data-tell-me_ver02\site"
    mkdocs_config_path = r"C:\gppj\data-tell-me_ver02\mkdocs.yml"

    # HTML로 변환 및 로컬 서버 시작
    convert_md_to_html(md_directory, html_directory, mkdocs_config_path)
    • 이 코드를 실행하면 Markdown 파일이 HTML로 변환되고, 로컬 서버가 시작됩니다. 변환된 HTML 파일은 html_directory에 저장됩니다.
