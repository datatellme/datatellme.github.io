# Load necessary libraries
library(rmarkdown)
library(reticulate)

# Set Python environment if needed (remove if not necessary)
use_python("C:/Users/oem/miniconda3/envs/myfirstenv", required = TRUE)

# Function to render the Rmd file to HTML
knit_rmd_to_html <- function(rmd_file, output_file = NULL) {
  tryCatch({
    if (!file.exists(rmd_file)) {
      stop(sprintf("Error: Rmd file does not exist -> %s", rmd_file))
    }
    
    # Render the Rmd file to HTML
    rmarkdown::render(input = rmd_file, output_file = output_file)
    
    # Success message
    message(sprintf("Successfully knitted %s to HTML.", rmd_file))
  }, error = function(e) {
    message(sprintf("An error occurred: %s", e$message))
  })
}

# Define output directories
output_directory <- "C:/gppj/econ_blog06/0101_git"
output_directory01 <- "C:/gppj/data-tell-me_ver05/quant/03_stock"

# Ensure the output directories exist
if (!dir.exists(output_directory)) {
  dir.create(output_directory, recursive = TRUE)
}
if (!dir.exists(output_directory01)) {
  dir.create(output_directory01, recursive = TRUE)
}

# Get current date-time
current_time <- format(Sys.time(), "%Y-%m-%d-%H")

# Define file paths
rmd_file <- file.path(output_directory, paste0(current_time, "-100종목 선정 방법_hompi_ver0712.Rmd"))
output_html <- file.path(output_directory, paste0(current_time, "-100종목 선정 방법_hompi_ver0712.html"))
output_html01 <- file.path(output_directory01, "index.html")

# Knit Rmd to HTML
knit_rmd_to_html(rmd_file, output_html)
knit_rmd_to_html(rmd_file, output_html01)
