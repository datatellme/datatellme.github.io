  • 여러 개의 mp4 파일을 읽어서 mp3로 변환하고 D:저장하는 파이썬 코드

  • tqdm 으로 변환 진행 상황을 진행바로 볼 수 있어요.

  • 오디오가 없는 mp4는 건너뛰고 경고를 출력합니다.

  • 저장할 mp3 이름은 원본 mp4 파일 이름을 따릅니다.

from moviepy.editor import VideoFileClip
import os
from tqdm import tqdm

# 원본 mp4 파일이 있는 폴더
path = r"C:\gppj\data\worship"

# 변환된 mp3 파일을 저장할 폴더
output_path = r"D:\music"

# Mp3 저장 폴더가 없다면 생성
os.makedirs(output_path, exist_ok=True)

# mp4 파일 목록 불러오기 및 정렬
file_list = sorted([f for f in os.listdir(path) if f.lower().endswith('.mp4')])

# 변환 시작
for file_name in tqdm(file_list, desc="MP4 to MP3 변환 중"):
    mp4_file = os.path.join(path, file_name)
    mp3_file = os.path.join(output_path, os.path.splitext(file_name)[0] + ".mp3")
    
    try:
        clip = VideoFileClip(mp4_file)
        if clip.audio is not None:
            clip.audio.write_audiofile(mp3_file)
            clip.close()
        else:
            print(f"오디오가 없는 파일입니다: {file_name}")
    except Exception as e:
        print(f"오류 발생: {file_name} - {e}")

print("✅ 모든 MP4 파일이 MP3로 변환되었습니다.")
