2 단계 작업

import requests
import os
from datetime import datetime
from tqdm import tqdm
from xml.etree import ElementTree as ET
import re

# 국립중앙도서관 API 키
API_KEY = "여기에_본인의_API_KEY_입력"

# BibTeX 파일 경로
output_dir = r"D:\APA"
os.makedirs(output_dir, exist_ok=True)
bibtex_file = os.path.join(output_dir, "book.bib")

def sanitize_bibtex_key(text):
    text = re.sub(r'\W+', '', text)
    return text.lower()

def get_book_info_nl(isbn):
    url = (
        f"http://www.nl.go.kr/app/nl/search/openApi/search.jsp?"
        f"key={API_KEY}&detailSearch=true&isbn={isbn}&apiType=xml"
    )
    response = requests.get(url)
    if response.status_code != 200:
        return None

    root = ET.fromstring(response.content)
    record = root.find(".//record")
    if record is None:
        return None

    return {
        "title": record.findtext("titleInfo/title", default="제목 없음"),
        "author": record.findtext("author", default="저자 없음"),
        "pub_year": record.findtext("pubYear", default="n.d."),
        "publisher": record.findtext("pubInfo", default="출판사 없음"),
        "isbn": isbn,
        "address": "Korea"
    }

def make_bibtex_entry(book):
    first_author_lastname = book["author"].split()[0].lower()
    key = sanitize_bibtex_key(first_author_lastname + book["pub_year"])

    return f"""@Book{{{key},
  title     = {{{book['title']}}},
  author    = {{{book['author']}}},
  publisher = {{{book['publisher']}}},
  address   = {{{book['address']}}},
  year      = {{{book['pub_year']}}},
  note      = {{ISBN {book['isbn']}}}
}}"""

def main():
    print("🔍 ISBN을 쉼표(,) 또는 줄바꿈으로 입력하세요:")
    raw_input = input("ISBN 목록 입력: ")
    isbn_list = [isbn.strip().replace("-", "") for isbn in raw_input.replace("\n", ",").split(",") if isbn.strip()]

    bib_entries = []

    print("\n📚 도서 정보 조회 중...")
    for isbn in tqdm(isbn_list, desc="진행 상황", unit="권"):
        try:
            book = get_book_info_nl(isbn)
            if book:
                bib = make_bibtex_entry(book)
                bib_entries.append(bib)
            else:
                print(f"{isbn}: ❌ 도서 정보를 찾을 수 없습니다.")
        except Exception as e:
            print(f"{isbn}: ⚠️ 오류 발생 - {e}")

    if bib_entries:
        with open(bibtex_file, "a", encoding="utf-8") as f:
            f.write("\n\n" + "\n\n".join(bib_entries))
        print(f"\n✅ BibTeX 항목이 누적 저장되었습니다:\n{bibtex_file}")
    else:
        print("❌ 저장할 항목이 없습니다.")

if __name__ == "__main__":
    main()

  • book.bib 파일이 이미 존재해도 덮어쓰지 않고 이어서 저장합니다.

  • open(…, “a”) 모드로 파일 열기 → 기존 내용 유지.

  • 각 실행 시 새 ISBN 도서 정보만 추가.
