  • 각 VideoFileClip 위에 TextClip을 만들어 CompositeVideoClip으로 합칩니다.

  • 자막은 클립 시작 0~2초 사이에 나타나도록 설정합니다.

  • 그 후 concatenate_videoclips()로 모두 이어붙입니다.

  • subtitles.txt 파일은 mp4 개수와 줄 수가 동일해야 가장 정확합니다.

  • 부족한 줄은 자동으로 Clip N: 파일명으로 대체됩니다.

  • sutf-8 인코딩을 사용하면 한글 자막도 문제 없이 적용됩니다.

from moviepy.editor import VideoFileClip, TextClip, CompositeVideoClip, concatenate_videoclips
import os
from tqdm import tqdm

# 원본 Mp4 파일이 있는 폴더
path = r"D:\data\moviepy\20230501"

# 합쳐진 Mp4 파일을 저장할 폴더
output_path = r"D:\data\moviepy\20230501_all"
os.makedirs(output_path, exist_ok=True)

# 자막 텍스트 파일 경로
subtitle_file = os.path.join(path, "subtitles.txt")

# Mp4 파일 목록 불러오기 및 정렬
file_list = sorted([f for f in os.listdir(path) if f.lower().endswith('.mp4')])

# 자막 텍스트 파일 읽기
if os.path.exists(subtitle_file):
    with open(subtitle_file, 'r', encoding='utf-8') as f:
        subtitles = [line.strip() for line in f.readlines()]
else:
    subtitles = []  # 자막 파일 없으면 빈 리스트

clips = []

print("📥 Mp4 파일 읽는 중 및 자막 적용 중...")
for idx, file_name in enumerate(tqdm(file_list, desc="MP4 처리 중")):
    mp4_file = os.path.join(path, file_name)
    try:
        base_clip = VideoFileClip(mp4_file)

        # 자막 텍스트 선택
        if idx < len(subtitles):
            subtitle_text = subtitles[idx]
        else:
            subtitle_text = f"Clip {idx+1}: {file_name}"

        # 자막 클립 생성
        txt_clip = TextClip(subtitle_text, fontsize=40, color='white',
                           font="C:/Windows/Fonts/NanumGothicBold.ttf",   # ← 한글 폰트   
                                    method='caption',               # ← 줄바꿈 가능
                                    size=(clip.w - 100, None) # ← 텍스트 최대 너비 지정 (영상 너비보다 작게)                         
                            )
        txt_clip = txt_clip.set_duration(2).set_position(('center', 'bottom'))

        # 자막과 비디오 클립 합성
        composite = CompositeVideoClip([base_clip, txt_clip.set_start(0)])
        clips.append(composite)

    except Exception as e:
        print(f"❌ 오류 발생: {file_name} - {e}")

# 모든 클립 이어붙이기
if clips:
    print("🛠️ 클립 이어붙이는 중...")
    final_clip = concatenate_videoclips(clips, method="compose")

    output_file = os.path.join(output_path, "all_with_subtitles.mp4")
    final_clip.write_videofile(output_file, codec="libx264", audio_codec="aac", preset="ultrafast")

    # 리소스 해제
    for clip in clips:
        clip.close()

    print("✅ 완료! 자막 포함 mp4가 저장되었습니다.")
else:
    print("⚠️ 클립이 존재하지 않거나 처리 중 오류가 발생했습니다.")
