from moviepy.editor import VideoFileClip, TextClip, CompositeVideoClip
import os
from tqdm import tqdm

# 원본 Mp4 파일이 있는 폴더
path = r"D:\data\moviepy\20230501_choice"

# 자막 포함 Mp4 파일 저장할 폴더
output_path = r"D:\data\moviepy\20230501_subtitled"
os.makedirs(output_path, exist_ok=True)

# 자막 텍스트 파일 경로
subtitle_file = os.path.join(path, "subtitles.txt")

# Mp4 파일 목록 불러오기 및 정렬
file_list = sorted([f for f in os.listdir(path) if f.lower().endswith('.mp4')])

# 자막 텍스트 읽기
if os.path.exists(subtitle_file):
    with open(subtitle_file, 'r', encoding='utf-8') as f:
        subtitles = [line.strip() for line in f.readlines()]
else:
    subtitles = []

print("🎬 각 클립에 자막 추가 및 개별 저장 중...")
for idx, file_name in enumerate(tqdm(file_list, desc="자막 처리 중")):
    mp4_file = os.path.join(path, file_name)
    try:
        clip = VideoFileClip(mp4_file)

        # 자막 텍스트
        if idx < len(subtitles):
            subtitle_text = subtitles[idx]
        else:
            subtitle_text = f"Clip {idx+1}: {file_name}"

        # 기본 자막 클립 (흰색)
        main_txt = TextClip(subtitle_text, fontsize=150, color='white',
                            font="C:/Windows/Fonts/NanumGothicBold.ttf",   # ← 한글 폰트   
                                    method='caption',               # ← 줄바꿈 가능
                                    size=(clip.w - 100, None) # ← 텍스트 최대 너비 지정 (영상 너비보다 작게)
         
        ).set_duration(clip.duration).set_position(('left', 'bottom'))

        # 그림자 자막 클립 (검정, 약간 아래/오른쪽)
#        shadow_txt = TextClip(subtitle_text, fontsize=150, color='black',
#                            font="C:/Windows/Fonts/NanumGothicBold.ttf",   # ← 한글 폰트   
#                                    method='caption',               # ← 줄바꿈 가능
#                                    size=(clip.w - 100, None) # ← 텍스트 최대 너비 지정 (영상 너비보다 작게)
#         ).set_duration(clip.duration).set_position(('left', clip.h - 100.05))  # 약간 아래 위치

        # 페이드 인/아웃 효과
        main_txt = main_txt.fadein(1).fadeout(1)
#        shadow_txt = shadow_txt.fadein(0.5).fadeout(0.5)

        # 자막 합성 (#그림자 → 텍스트 → 영상)
        composite = CompositeVideoClip([clip, main_txt]) # 약간 아래 위치: shadow_txt, 

        # 저장 파일명
        output_filename = os.path.splitext(file_name)[0] + "_subtitled.mp4"
        output_file = os.path.join(output_path, output_filename)

        # 저장
        composite.write_videofile(output_file, codec="libx264", audio_codec="aac", preset="ultrafast")
        composite.close()
        clip.close()

    except Exception as e:
        print(f"❌ 오류 발생: {file_name} - {e}")

print("✅ 모든 파일이 자막 포함 개별 영상으로 저장되었습니다.")
