def get_book_info_openlibrary(isbn):
    url = f"https://openlibrary.org/api/books?bibkeys=ISBN:{isbn}&format=json&jscmd=data"
    response = requests.get(url)
    if response.status_code != 200:
        return None

    data = response.json()
    book_data = data.get(f"ISBN:{isbn}")
    if not book_data:
        return None

    title = book_data.get("title", "Unknown Title")
    authors = [author["name"] for author in book_data.get("authors", [])]
    author = " and ".join(authors) if authors else "Unknown Author"
    publishers = [pub["name"] for pub in book_data.get("publishers", [])]
    publisher = publishers[0] if publishers else "Unknown Publisher"
    year = book_data.get("publish_date", "n.d.")
    year = re.findall(r'\d{4}', year)
    year = year[0] if year else "n.d."

    return {
        "title": title,
        "author": author,
        "publisher": publisher,
        "year": year,
        "isbn": isbn,
        "address": "unknown"
    }
