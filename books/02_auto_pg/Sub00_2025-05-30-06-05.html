  • 아래는 ISBN을 입력하면 온라인에서 도서 정보를 조회한 후, APA 스타일의 참고문헌(bibliography)을 생성해주는 Python 프로그램입니다.

  • 아래는 여러 개의 ISBN을 한 번에 입력받아 각각에 대해 도서 정보를 조회하고, APA 스타일로 포맷된 참고문헌 리스트를 D:폴더에 텍스트 파일로 저장하는 파이썬 코드입니다.

  • 이 프로그램은 Open Library API를 사용하여 ISBN으로 책 정보를 검색합니다.

  • 9791187444886처럼 국내 ISBN(979-로 시작)은 Open Library API에서 종종 정보를 제공하지 않아 “도서 정보를 찾을 수 없습니다” 메시지가 뜹니다.

  • 이 API는 주로 북미/국제 출판 데이터에 특화되어 있기 때문에 국내 서적 데이터는 빈약합니다.

import requests
import os
from datetime import datetime
from tqdm import tqdm
import re

# BibTeX 파일 저장 경로
output_dir = r"D:\APA"
os.makedirs(output_dir, exist_ok=True)
bibtex_file = os.path.join(output_dir, "book.bib")

# BibTeX key 정리용
def sanitize_bibtex_key(text):
    return re.sub(r'\W+', '', text).lower()

# Open Library API에서 책 정보 조회
def get_book_info_openlibrary(isbn):
    url = f"https://openlibrary.org/api/books?bibkeys=ISBN:{isbn}&format=json&jscmd=data"
    response = requests.get(url)
    if response.status_code != 200:
        return None

    data = response.json()
    book_data = data.get(f"ISBN:{isbn}")
    if not book_data:
        return None

    title = book_data.get("title", "Unknown Title")
    authors = [author["name"] for author in book_data.get("authors", [])]
    author = " and ".join(authors) if authors else "Unknown Author"
    publishers = [pub["name"] for pub in book_data.get("publishers", [])]
    publisher = publishers[0] if publishers else "Unknown Publisher"
    year = book_data.get("publish_date", "n.d.")
    year = re.findall(r'\d{4}', year)
    year = year[0] if year else "n.d."

    return {
        "title": title,
        "author": author,
        "publisher": publisher,
        "year": year,
        "isbn": isbn,
        "address": "unknown"
    }

# BibTeX 항목 만들기
def make_bibtex_entry(book):
    first_author_lastname = book["author"].split()[0].lower()
    key = sanitize_bibtex_key(first_author_lastname + book["year"])

    return f"""@Book{{{key},
  title     = {{{book['title']}}},
  author    = {{{book['author']}}},
  publisher = {{{book['publisher']}}},
  address   = {{{book['address']}}},
  year      = {{{book['year']}}},
  note      = {{ISBN {book['isbn']}}}
}}"""

# 메인 실행
def main():
    print("🔍 ISBN을 쉼표(,) 또는 줄바꿈으로 입력하세요:")
    raw_input = input("ISBN 목록 입력: ")
    isbn_list = [isbn.strip().replace("-", "") for isbn in raw_input.replace("\n", ",").split(",") if isbn.strip()]

    bib_entries = []

    print("\n📚 도서 정보 조회 중...")
    for isbn in tqdm(isbn_list, desc="진행 상황", unit="권"):
        try:
            book = get_book_info_openlibrary(isbn)
            if book:
                bib = make_bibtex_entry(book)
                bib_entries.append(bib)
            else:
                print(f"{isbn}: ❌ 도서 정보를 찾을 수 없습니다.")
        except Exception as e:
            print(f"{isbn}: ⚠️ 오류 발생 - {e}")

    if bib_entries:
        with open(bibtex_file, "a", encoding="utf-8") as f:
            f.write("\n\n" + "\n\n".join(bib_entries))
        print(f"\n✅ BibTeX 항목이 누적 저장되었습니다:\n{bibtex_file}")
    else:
        print("❌ 저장할 항목이 없습니다.")

if __name__ == "__main__":
    main()
