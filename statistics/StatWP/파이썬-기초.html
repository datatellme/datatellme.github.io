1.2.2.2 배열 연산하기

import numpy as np

arr1 = np.array([10, 15, 20, 30]) 
arr2 = np.array([1, 1, 2, 3])

print('----------덧셈----------')
## ----------덧셈----------
print(arr1+arr2)
## [11 16 22 33]
print('----------뺄셈----------')
## ----------뺄셈----------
print(arr1-arr2)
## [ 9 14 18 27]
print('----------곱셈----------')
## ----------곱셈----------
print(arr1*arr2)
## [10 15 40 90]
print('----------나눗셈--------')
## ----------나눗셈--------
print(arr1/arr2)
## [10. 15. 10. 10.]

1.2.2.3 배열의 인덱싱과 슬라이싱

  • 인덱싱 (Indexing)
import numpy as np

a1 = np.array([0, 10, 20, 30, 40, 50])
a1
## array([ 0, 10, 20, 30, 40, 50])
a1[4]
## 40
a1[5] = 70 
a1
## array([ 0, 10, 20, 30, 40, 70])
a1[[1,3,4]] 
## array([10, 30, 40])
  • 슬라이싱 (Slicing)
import numpy as np

b1 = np.array([0, 1, 2, 3, 4, 5])
b1
## array([0, 1, 2, 3, 4, 5])
b1[1:4]
## array([1, 2, 3])
b1[2:]
## array([2, 3, 4, 5])
b1[2:5] = np.array([25, 35, 45])
b1
## array([ 0,  1, 25, 35, 45,  5])
